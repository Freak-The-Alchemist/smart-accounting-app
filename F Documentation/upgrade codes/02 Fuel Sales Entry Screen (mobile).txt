// mobile/src/screens/FuelSaleForm.tsx
import React, { useState } from 'react';
import { View, Text, TextInput, Button, StyleSheet, Alert } from 'react-native';
import { collection, addDoc, Timestamp } from 'firebase/firestore';
import { db, auth } from '@smart-accounting/shared/firebase';
import { FUEL_TYPES, PAYMENT_METHODS } from '@smart-accounting/shared/constants';

const FuelSaleForm = () => {
  const [litersSold, setLitersSold] = useState('');
  const [pricePerLiter, setPricePerLiter] = useState('');
  const [fuelType, setFuelType] = useState('Petrol');
  const [paymentMethod, setPaymentMethod] = useState('Cash');
  const [pumpId, setPumpId] = useState('1');

  const handleSubmit = async () => {
    const user = auth.currentUser;
    if (!user) return alert('Not logged in');

    const liters = parseFloat(litersSold);
    const price = parseFloat(pricePerLiter);
    if (isNaN(liters) || isNaN(price)) return alert('Enter valid numbers');

    const totalAmount = liters * price;

    try {
      await addDoc(collection(db, 'fuelSales'), {
        date: Timestamp.now().toDate().toISOString(),
        pumpId,
        fuelType,
        litersSold: liters,
        pricePerLiter: price,
        totalAmount,
        paymentMethod,
        attendantId: user.uid
      });
      Alert.alert('Success', 'Fuel sale recorded.');
      setLitersSold('');
      setPricePerLiter('');
    } catch (err) {
      console.error(err);
      Alert.alert('Error', 'Could not save sale.');
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.label}>Fuel Type</Text>
      <TextInput value={fuelType} onChangeText={setFuelType} style={styles.input} />

      <Text style={styles.label}>Pump ID</Text>
      <TextInput value={pumpId} onChangeText={setPumpId} style={styles.input} />

      <Text style={styles.label}>Liters Sold</Text>
      <TextInput value={litersSold} onChangeText={setLitersSold} keyboardType="numeric" style={styles.input} />

      <Text style={styles.label}>Price per Liter</Text>
      <TextInput value={pricePerLiter} onChangeText={setPricePerLiter} keyboardType="numeric" style={styles.input} />

      <Text style={styles.label}>Payment Method</Text>
      <TextInput value={paymentMethod} onChangeText={setPaymentMethod} style={styles.input} />

      <Button title="Record Sale" onPress={handleSubmit} />
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 20 },
  label: { marginTop: 10 },
  input: {
    borderWidth: 1,
    borderColor: '#ccc',
    padding: 8,
    marginTop: 4,
    borderRadius: 4
  }
});

export default FuelSaleForm;
